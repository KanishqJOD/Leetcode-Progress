class Solution {
    int rec(int i,int j,vector<int>& cuts, vector<vector<int>>&dp) {
        if(i > j) {
            return 0;
        }
        if(dp[i][j] != -1) {
            return dp[i][j];
        }
        int mini = INT_MAX;
        for(int g = i; g <=j;g++) {
            int cost = cuts[j + 1] - cuts[i-1] +
            rec(i,g-1,cuts,dp) + rec(g + 1,j,cuts,dp);
            mini = min(mini,cost);
        }
        return dp[i][j] = mini;
    }
